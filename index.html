Quick Start

From install to inference in under 60 seconds.

Detection 
# Install
pip install flashdet

# Python API
from flashdet import Predictor

detector = Predictor(pretrained_coco=True)
results = detector.predict("image.jpg")

for cls, score, x1, y1, x2, y2 in results:
    print(f"{cls}: {score:.2f}")
Training 
# Fine-tune with LoRA on custom data
from flashdet import Trainer

trainer = Trainer(
    pretrained_coco=True,
    lora=True,
    epochs=50,
    train_images="data/train",
    val_images="data/val",
)
trainer.train()
FlashStudio UI 
# Install FlashStudio
pip install flashstudio

# Launch interactive dashboard
flashstudio

# Opens at http://localhost:8501

CLI 
# Train from command line
flashdet train --pretrained-coco --lora --epochs 50

# Export to ONNX
flashdet export --format onnx --half

# Run benchmark
flashdet benchmark --model m --input-size 320

# Download dataset
flashdet download --dataset coco128
