Quick Start

From install to inference in 60 seconds.

detect.py 
from flashdet import Predictor

# Load COCO-pretrained model (auto-downloads weights)
detector = Predictor(pretrained_coco=True)

# Run inference on image
detections = detector.predict("street.jpg")

for cls_name, score, x1, y1, x2, y2 in detections:
    print(f"{cls_name}: {score:.2f} [{x1},{y1},{x2},{y2}]")

# Process video with live visualization
detector.predict_video("traffic.mp4", output_dir="results/")
train.py 
from flashdet import Trainer

# Fine-tune with LoRA on custom dataset
trainer = Trainer(
    pretrained_coco=True,
    lora=True,
    lora_rank=8,
    epochs=50,
    batch_size=32,
    train_images="data/train",
    val_images="data/val",
    amp=True,
)

# Starts training (~5 min on single GPU)
results = trainer.train()
print(f"Best mAP@0.5: {results['best_map50']:.4f}")
terminal 
# Install any FlashVision package
$ pip install flashdet flashseg flashcls flashocr flashtrack

# Train object detector with LoRA
$ flashdet train --pretrained-coco --lora --epochs 50

# Run segmentation
$ flashseg predict --source image.jpg --model m

# Export to ONNX for edge deployment
$ flashdet export --format onnx --half

# Benchmark model speed
$ flashdet benchmark --model m --input-size 320
